sub _edit {
  my $self = shift;
  my %options = @_;

  # load template for edit screen
  my $template = $self->load_tmpl('edit.tmpl');

  # include reply_id in hidden field if replying
  $template->param(reply_id => $reply_id) if exists $options{reply_id};
  
  # return template output
  return $template->output;
}
