var season = 'first';

Fix the code to get ‘Spartacus: Blood and Sand’:

var season = 'first';
var result = ('Spartacus: ' + season === 'first' ? 'Blood and Sand' : 'Gods of the Arena');
var season = 'first';
var result = 'Spartacus: ' + (season === 'first' ? 'Blood and Sand' : 'Gods of the Arena');
assert(result === 'Spartacus: Blood and Sand');
